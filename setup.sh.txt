#!/bin/bash
set -e  # Exit on error

echo "🚀 Setting up DOTFILES..."

# Update system
echo "📦 Updating system packages..."
sudo apt update && sudo apt upgrade -y

# Install packages
echo "📦 Installing required packages..."
PACKAGES=(
    vim
    gh
    curl
    wget
    btop
    neovim
    fzf
    eza
    build-essential
    python3
    python3-pip
    python3-venv
    zsh
)

sudo apt install -y "${PACKAGES[@]}"
echo "✅ Packages installed successfully."

# Dotfiles setup
DOTFILES_DIR="$HOME/dotfiles"
REPO_URL="https://github.com/MegLv/dotfiles.git"

echo "📥 Setting up dotfiles..."
if [ -d "$DOTFILES_DIR" ]; then
    echo "Repository exists, pulling latest changes..."
    cd "$DOTFILES_DIR"
    git pull origin main || echo "⚠️ Pull failed, continuing anyway..."
else
    echo "Cloning repository..."
    git clone "$REPO_URL" "$DOTFILES_DIR"
fi

# Oh My Zsh
echo "🔧 Setting up Oh My Zsh..."
if [ ! -d "$HOME/.oh-my-zsh" ]; then
    echo "Installing Oh My Zsh..."
    sh -c "$(curl -fsSL https://raw.github.com/ohmyzsh/ohmyzsh/master/tools/install.sh)" "" --unattended
else
    echo "Oh My Zsh already installed."
fi

ZSH_CUSTOM="${ZSH_CUSTOM:-$HOME/.oh-my-zsh/custom}"

# Install Zsh plugins
mkdir -p "$ZSH_CUSTOM/plugins"
for plugin in zsh-autosuggestions zsh-syntax-highlighting; do
    if [ ! -d "$ZSH_CUSTOM/plugins/$plugin" ]; then
        echo "Installing $plugin..."
        git clone --depth=1 "https://github.com/zsh-users/$plugin" "$ZSH_CUSTOM/plugins/$plugin"
    else
        echo "✅ $plugin already installed."
    fi
done

# Install Powerlevel10k
P10K_DIR="$ZSH_CUSTOM/themes/powerlevel10k"
if [ ! -d "$P10K_DIR" ]; then
    echo "Installing Powerlevel10k..."
    git clone --depth=1 https://github.com/romkatv/powerlevel10k.git "$P10K_DIR"
else
    echo "✅ Powerlevel10k already installed."
fi

# Symbolic links
echo "🔗 Creating symbolic links..."
create_symlink() {
    local source="$1"
    local target="$2"
    if [ -e "$target" ] && [ ! -L "$target" ]; then
        echo "Backing up: $target → ${target}.backup"
        mv "$target" "${target}.backup"
    fi
    ln -sf "$source" "$target"
    echo "✅ Linked: $target"
}

# Link main config files (add more as your repo grows)
create_symlink "$DOTFILES_DIR/.zshrc" "$HOME/.zshrc"
# create_symlink "$DOTFILES_DIR/.p10k.zsh" "$HOME/.p10k.zsh"   # Uncomment when you add it
# create_symlink "$DOTFILES_DIR/nvim" "$HOME/.config/nvim"

# Change default shell to Zsh
if [[ "$SHELL" != *"zsh"* ]]; then
    echo "🔄 Changing default shell to Zsh..."
    chsh -s "$(which zsh)"
fi

echo ""
echo "🎉 Setup completed successfully!"
echo "Please restart your terminal or run:   exec zsh"
